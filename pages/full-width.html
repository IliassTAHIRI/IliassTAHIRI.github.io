Interpolation in Python¶

In this first post I present some of the interpolation techniques used in python more specifically scipy.interpolation.interp1d.

In [52]:
import numpy as np
from scipy.interpolate import interp1d
import matplotlib.pylab as plt


# data points
xmax=3
n=15
x=np.linspace(0,xmax,n)
y=np.exp(-x**2) * np.sin(2*np.pi*x)

# Interpolation
f_nearest = interp1d(x, y, kind='nearest')
f_linear  = interp1d(x, y)
f_cubic   = interp1d(x, y, kind='cubic')

x2=np.linspace(0,xmax,n*100)

#Plotting
plt.plot(x2, f_nearest(x2), label='Nearest')
plt.plot(x2, f_linear(x2), label='Linear')
plt.plot(x2, f_cubic(x2), label='Cubic')
plt.plot(x,y,'o', label='Data points')
plt.legend()
plt.show()

from scipy.interpolate import interp2d

# 2D space - Grid

x = np.linspace(0, 10, 20)
y = np.linspace(0, 10, 20)
X, Y = np.meshgrid(x, y)
Z = np.sin(np.pi*X) * np.exp(-Y)
f = interp2d(x, y, Z, kind='cubic')

x2 = np.linspace(0, 4, 100)
y2 = np.linspace(0, 4, 100)
Z2 = f(x2, y2)



fig, ax = plt.subplots(nrows=1, ncols=2)
ax[0].pcolormesh(X, Y, Z)

X2, Y2 = np.meshgrid(x2, y2)
ax[1].pcolormesh(X2, Y2, Z2)
plt.show()
